214 // The problem with using a world coordinate as the variable when triangulating is that the solver
215 // might overshoot and end up behind you. This happens a lot if the initial estimate is e.g. 1000 m
216 // away: The solver realizes it is way too far and follows the gradient back close to zero. Very
217 // often it will blow past zero and end up behind you
218 
219 // The trick to fix this is to use inv = world / |world|^2 as the variable instead. Using
220 // disparity = 1 / |world|,
221 // it can be seen that
222 // inv = disparty * unit(world),
223 // so this accomplishes two things:
224 // - The variable is proportional to disparity. We care about pixels, not meters
225 // - To end up behind you, the solver needs to cross through infinity (hard) instead of zero (easy)
226 struct TriangulationFunctor {
227  static ceres::CostFunction* addResidual(
228  ceres::Problem& problem,
229  Camera::Vector3& inv,
230  const Camera& camera,
231  const Camera::Vector2& pixel,
232  const bool robust = false) {
233  auto* cost = new CostFunction(new TriangulationFunctor(camera, pixel));
234  auto* loss = robust ? new ceres::HuberLoss(1.0) : nullptr;
235  problem.AddResidualBlock(cost, loss, inv.data());
236  return cost;
237  }

239  bool operator()(double const* const inv, double* residuals) const {
240  Eigen::Map<const Camera::Vector3> i(inv);
241  Eigen::Map<Camera::Vector2> r(residuals);
242 
243  Camera::Vector3 w = i / i.squaredNorm();
244 
245  // transform world with camera and compare to pixel
246  r = camera.pixel(w) - pixel;

248  return true;

275 Camera::Vector3 triangulateNonlinear(const Observations& observations, const bool forceInFront) {
276  CHECK_GE(observations.size(), 2);
277  ceres::Solver::Options options;
278  options.max_num_iterations = 10;
279 
280  // initial value is average of distant points
281  const Camera::Real kInitialDistance = 10; // 10 meters, not hugely important
282  Camera::Vector3 world = averageAtDistance(observations, kInitialDistance);
283  Camera::Vector3 inv = world / world.squaredNorm();
284 
285  ceres::Problem problem;
286  for (const auto& obs : observations) {
287  TriangulationFunctor::addResidual(problem, inv, obs.first, obs.second);
288  }
289 
290  ceres::Solver::Summary summary;
291  ceres::Solve(options, &problem, &summary);
292 
293  world = inv / inv.squaredNorm();
294 
295  if (forceInFront) {
296  for (const auto& obs : observations) {
297  if (obs.first.isBehind(world)) {
298  return averageAtDistance(observations, Camera::kNearInfinity);
299  }
300  }
301  }
302 
303  return world;
304 }
