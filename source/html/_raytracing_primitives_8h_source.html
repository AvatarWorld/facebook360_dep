58 RayIntersectionResult rayIntersectTriangle(const Ray& ray, const Triangle& tri) {
59  const cv::Vec3f q = ray.dir.cross(tri.e2);
60  const float a = tri.e1.dot(q);
61 
62  // avoid near-parralel result (for numerical reasons)
63  if (a * a < 0.0001f) {
64  return RayIntersectionResult::miss();
65  }
66 
67  const cv::Vec3f s = (ray.origin - tri.v0) / a;
68  const cv::Vec3f r = s.cross(tri.e1);
69 
70  // barycentric coordinates
71  const float b0 = s.dot(q);
72  const float b1 = r.dot(ray.dir);
73  const float b2 = 1.0f - b0 - b1;
74 
75  if (b0 < 0.0f || b1 < 0.0f || b2 < 0.0f) {
76  return RayIntersectionResult::miss();
77  }
78 
79  const float intersectionDist = tri.e2.dot(r);
80  if (intersectionDist < 0.0f) {
81  return RayIntersectionResult::miss();
82  }
83 
84  return RayIntersectionResult(true, intersectionDist, tri.selfIdx);
85 }
