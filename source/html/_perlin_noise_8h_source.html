54 static inline float grad(int hash, float x, float y, float z) {
55  int h = hash & 15;
56  float u = h < 8 ? x : y;
57  float v = h < 4 ? y : h == 12 || h == 14 ? x : z;
58  return ((h & 1) == 0 ? u : -u) + ((h & 2) == 0 ? v : -v);
59 }

62  int X = (int)floor(x) & 255;
63  int Y = (int)floor(y) & 255;
64  int Z = (int)floor(z) & 255;
65  x -= floor(x);
66  y -= floor(y);
67  z -= floor(z);
68  float u = fade(x);
69  float v = fade(y);
70  float w = fade(z);

72  int AA = p[A] + Z;
