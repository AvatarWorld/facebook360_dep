86  void readBegin(PendingRead& pending, const std::vector<Segment>& segments, uint64_t offset)
87  const {
88  pending.handle = handle;
89  pending.overlapped.Offset = offset & 0xFFFFFFFF;
90  pending.overlapped.OffsetHigh = offset >> 32;
91  pending.overlapped.hEvent = CreateEvent(nullptr, TRUE, FALSE, nullptr);
92  pending.overlapped.Internal = 0;
93  pending.overlapped.InternalHigh = 0;
94  // compute the total number of bytes
95  uint64_t total = 0;
96  for (const auto& segment : segments) {
97  uint8_t* dst = segment.iov_base;
98  uint64_t size = segment.iov_len;
99  CHECK(align(dst, kPageSize) == dst);
100  CHECK(align(size, kPageSize) == size);
101  total += segment.iov_len;
102  }
103  // create a FILE_SEGMENT_ELEMENT per page plus an extra for NULL
104  std::vector<FILE_SEGMENT_ELEMENT> pages(total / kPageSize + 1);
105  size_t pageIndex = 0;
106  for (const auto& segment : segments) {
107  uint8_t* dst = segment.iov_base;
108  uint64_t size = segment.iov_len;
109  for (uint64_t i = 0; i < size / kPageSize; ++i) {
110  pages[pageIndex].Buffer = dst;
111  ++pageIndex;
112  dst += kPageSize;
113  }
114  }
115  CHECK_EQ(pageIndex, pages.size() - 1);
116  pages.back().Buffer = nullptr;

118  const BOOL ok = ReadFileScatter(

120  if (!ok) {

189  // keep a decoder ring to map filehandles to filenames
190  std::map<HANDLE, std::string> filenames;
191  // a request is identified by filehandle, offset
192  using Request = std::pair<HANDLE, uint64_t>;
193  // the events associated with a request are just 3 timestamps
194  static const size_t kEventCount = 3; // begin, get, done
195  using Clock = std::chrono::high_resolution_clock;
196  using Events = std::array<Clock::time_point, kEventCount>;
197  // live requests are stored in a map, completed requests in a vector
198  std::map<Request, Events> live;
199  std::vector<std::pair<Request, Events>> completed;
200  // protect updates with a mutex
201  std::mutex mutex;
202 
203  // call to populate filehandle -> filename decoder ring
204  void addFile(HANDLE filehandle, const std::string& filename) {
205  std::lock_guard<std::mutex> guard(mutex);
206  filenames[filehandle] = filename;
207  }
208 
209  // call when event occurs
210  void event(HANDLE filehandle, uint64_t offset, size_t index) {
211  CHECK_LT(index, kEventCount);
212  Request request(filehandle, offset);
213  Clock::time_point now = Clock::now();
214  std::lock_guard<std::mutex> guard(mutex);
215  live[request][index] = now;
216  if (index + 1 == kEventCount) { // last event, transfer from live to log
217  completed.emplace_back(request, live[request]);
218  live.erase(request);
219  }
220  }
221 
222  void dump(const std::string& filename) {
223  std::ofstream file(filename);
224  for (const auto& entry : completed) {
225  file << "\"" << filenames[entry.first.first] << "\"";
226  file << "\t" << entry.first.second;
227  for (const auto& timestamp : entry.second) {
228  // offset timestamps from first timestamp
229  static Clock::time_point offset = timestamp;
230  std::chrono::duration<double> elapsed = timestamp - offset;
231  file << "\t" << elapsed.count();
232  }
233  file << std::endl;
234  }
235  }
