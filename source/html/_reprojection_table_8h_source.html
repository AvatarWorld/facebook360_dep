14 // a reprojection table tells you - for each pixel in dst - where in src to look
15 // i.e. it implements
16 // f(xy, disparity) = src.pixel(dst.rig(xy, 1 / disparity))
17 
18 // the underlying implementation is a 3D piecewise linear table with enough
19 // points to stay within tolerance of the correct answer
20 
21 // the reprojection table is a 3D table with dimensions shape. the first entry
22 // represents
23 // (0 - margin.x, 0 - margin.y, min disparity)
24 
25 // the last entry represents
26 // (1 + margin.x, 1 + margin.y, max disparity)
27 
28 // the reprojection table is meant to be used as a 3D texture. textures are
29 // addressed with texture coordinates that range from 0 at the *outside* corner
30 // of the first texel to 1 at the outside corner of the last texel
31 
32 // that means that the value represented by the first entry must be remapped to
33 // the texture coordinate of the center of the first texel and vice versa:
34 // 0 + texel / 2 ... 1 - texel / 2
35 
36 // scale and offset perform this remapping:
37 // texture coordinates = input * scale + offset
38 
39 // there is another similar coordinate system used internally while building the
40 // table, called normalized coordinates. they are 0 at the center of the first
41 // texel and 1 at the center of the last texel. so similar to texture coors but
42 // without the half-texel adjustment
43 
44 class ReprojectionTable {
45  public:
46  ReprojectionTable(
47  const Camera& dst,
48  const Camera& src,
49  const Camera::Vector2& tolerance,
50  const Camera::Vector2& margin = {0, 0})
51  : margin(margin) {
52  CHECK(dst.isNormalized());
53  if (src.overlap(dst) == 0) {
54  shape.setOnes();
55  values = {{-1, -1}}; // outside
56  return;
57  }
58 
59  // compute the resolution required in each dimension
60  for (int dim = 0; dim < shape.size(); ++dim) {
61  // start by checking error at kN^3 cells, increasing end[dim] as needed
62  static const int kN = 10;
63  static const float kFactor = 1.2f;
64  for (IndexType end = IndexType::Constant(kN);; end[dim] *= kFactor) {
65  if (isWithinTolerance(dst, src, end, dim, tolerance, margin)) {
66  shape[dim] = end[dim] + 1;
67  break;
68  }
69  }
70  }
71 
72  // now create table with the computed shape
73  values.resize(shape.prod());
74  for (IndexType i(0, 0, 0); good(i, shape); increment(i, shape)) {
75  const Camera::Vector3 normalized = divide(i, shape - 1);
76  values[flatten(i, shape)] = compute(dst, src, normalized, margin);
77  }
78  }

87  Eigen::Array3f getScale() const {
88  // input range
89  Eigen::Array3f input(1 + 2 * margin.x(), 1 + 2 * margin.y(), maxDisparity() - minDisparity());
90  // output range
91  Eigen::Array3f output = 1 - 1 / shape.cast<float>();
92  return output / input;
93  }
94 
95  Eigen::Array3f getOffset() const {
96  // input
97  Eigen::Array3f input(-margin.x(), -margin.y(), minDisparity());
98  // output
99  Eigen::Array3f output = 0.5f / shape.cast<float>();
100  // output = input * scale + offset <=>
101  return output - input * getScale();
102  }

104  // do a trilinear lookup in the table
105  // note: this is slow. use a GPU, that is what they are for
106  Entry lookup(const Entry& xy, const float disparity) const {
107  Eigen::Array3f texcoor = getScale() * Eigen::Array3f(xy.x(), xy.y(), disparity) + getOffset();
108  Eigen::Array3f unnorm = texcoor * shape.cast<float>();
109 
110  // accumulate entries from a 2^3 cube beginning with floor(unnorm - 0.5)
111  IndexType begin = floor(unnorm - 0.5).cast<int>();
112  CHECK((0 <= begin && begin < shape - 1).all()) << begin << "out of range";
113  Entry result = {0, 0};
114  for (int z = 0; z < 2; ++z) {
115  for (int y = 0; y < 2; ++y) {
116  for (int x = 0; x < 2; ++x) {
117  IndexType index = begin + IndexType(x, y, z);
118  Eigen::Array3f diff = (index.cast<float>() + 0.5f - unnorm).abs();
119  float weight = (1 - diff).prod();
120  CHECK(0 <= weight && weight <= 1) << weight;
121  result += weight * values[flatten(index, shape)];
122  }
123  }
124  }
125  return result;
126  }

156  static float normalizeDisparity(float disparity) {
157  // remap disparity from minDisparity() ... maxDisparity() to 0 ... 1
158  return (disparity - minDisparity()) / (maxDisparity() - minDisparity());

160  static float unnormalizeDisparity(float normalized) {
161  // remap normalized from 0 ... 1 to minDisparity() ... maxDisparity()
162  CHECK(0 <= normalized && normalized <= 1);
163  return (1 - normalized) * minDisparity() + normalized * maxDisparity();

171  CHECK(0 <= normalized.x() && normalized.x() <= 1);

201  static bool isWithinTolerance(
202  const Camera& dst,
203  const Camera& src,
204  const IndexType& end,
205  int dim,
206  const Camera::Vector2& tolerance,
207  const Camera::Vector2& margin) {
208  const Eigen::Array2f tol = tolerance.array().cast<float>();
209  for (IndexType i(0, 0, 0); good(i, end); increment(i, end)) {
210  // compute values in center of cell
211  Camera::Vector3 normalized = divide(i, end, 0.5);
212  const Camera::Vector2 xy = unnormalizeXY(normalized, margin);
213  if (!dst.isOutsideImageCircle(xy)) {
214  const float disparity = unnormalizeDisparity(normalized.z());
215  Camera::Vector2 exact;
216  if (src.sees(dst.rig(xy, 1 / disparity), exact)) {
217  // compute sample on either side of normalized along dimension dim
218  normalized[dim] -= 0.5 / end[dim];
219  Entry lo = compute(dst, src, normalized, margin);
220  normalized[dim] += 1.0 / end[dim];
221  Entry hi = compute(dst, src, normalized, margin);
222 
223  // does sub-texel precision error exceed tolerance?
224  const float kSubtexelPrecision = 1.0f / 512;
225  Eigen::Array2f sub = (hi - lo).array() * kSubtexelPrecision;
226  if ((abs(sub.array()) > tol).any()) {
227  return false; // error exceeds tolerance
228  }
229 
230  // does linear approximation error exceed tolerance?
231  Entry lin = (lo + hi) / 2 - exact.cast<float>();
232  if ((abs(lin.array()) > tol).any()) {
233  return false; // error exceeds tolerance
234  }
235  }
236  }
237  }
238  return true;
239  }
