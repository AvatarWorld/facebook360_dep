199  // estimate the fraction of the frame that is covered by the other camera
200  Real overlap(const Camera& other) const {
201  // just brute force probeCount x probeCount points
202  const int kProbeCount = 10;
203  int inside = 0;
204  for (int y = 0; y < kProbeCount; ++y) {
205  for (int x = 0; x < kProbeCount; ++x) {
206  const Vector2 p = Vector2(x, y).cwiseProduct(resolution) / (kProbeCount - 1);
207  if (!isOutsideImageCircle(p) && other.sees(rigNearInfinity(p))) {
208  ++inside;
209  }
210  }
211  }
212  return inside / Real(kProbeCount * kProbeCount);
213  }

256  Real undistort(const Real y) const {
257  if (getDistortion().isZero()) {
258  return y; // short circuit common case
259  }
260 
261  if (y >= distort(distortionMax_)) {

263  }

265  const Real smidgen = 1.0 / kNearInfinity;
266  const int kMaxSteps = 10;
267 
268  // solve y = distort(x) for y using newton's method
269  Real x0 = 0;
270  Real y0 = 0;
271  Real dy0 = 1;
272  for (int step = 0; step < kMaxSteps; ++step) {
273  const Real x1 = (y - y0) / dy0 + x0;
274  const Real y1 = distort(x1);
275  if (std::abs(y1 - y) < smidgen) {
276  return x1; // close enough
277  }
278  Real dy1 = (distort(x1 + smidgen) - y1) / smidgen;
279  CHECK_GE(dy1, 0) << "went past a maximum";
280  x0 = x1;
281  y0 = y1;
282  dy0 = dy1;
283  }
284  return x0; // this should not happen
285  }

302  Vector2 cameraToSensor(const Vector3& camera) const {
303  // FTHETA: r = theta
304  // RECTILINEAR: r = tan(theta)
305  // EQUISOLID: r = 2 sin(theta / 2)
306  // ORTHOGRAPHIC: r = sin(theta)
307  // see https://wiki.panotools.org/Fisheye_Projection
308  if (type == Type::FTHETA) {
309  Real xy = camera.head<2>().norm();
310  // r = theta <=>
311  // r = atan2(|xy|, -z)
312  Real r = atan2(xy, -camera.z());
313  return distort(r) / xy * camera.head<2>();
314  } else if (type == Type::RECTILINEAR) {
315  // r = tan(theta) <=>
316  // r = |xy| / -z <=>
317  // pre-distortion result is xy / -z
318  Vector2 pre = camera.head<2>() / std::max(0.0, -camera.z());
319  return distortFactor(pre.squaredNorm()) * pre;
320  } else if (type == Type::EQUISOLID) {
321  Real xy = camera.head<2>().norm();
322  // r = 2 sin(theta / 2) <=>
323  // using sin(theta / 2) = sqrt((1 - cos(theta)) / 2)
324  // r = 2 sqrt((1 + z / |xyz|) / 2)
325  Real r = 2 * sqrt((1 + camera.z() / camera.norm()) / 2);
326  return distort(r) / xy * camera.head<2>();
327  } else {
328  CHECK(type == Type::ORTHOGRAPHIC) << "unexpected: " << int(type);
329  // r = sin(theta) <=>
330  // r = |xy| / |xyz| <=>
331  // pre-distortion result is xy / |xyz|
332  Vector2 pre =
333  camera.z() < 0 ? camera.head<2>() / camera.norm() : camera.head<2>().normalized();
334  return distortFactor(pre.squaredNorm()) * pre;
335  }
336  }

338  // compute unit vector in camera coors from normalized sensor coors
339  Vector3 sensorToCamera(const Vector2& sensor) const {
340  // FTHETA: r = theta
341  // RECTILINEAR: r = tan(theta)
342  // EQUISOLID: r = 2 sin(theta / 2)
343  // ORTHOGRAPHIC: r = sin(theta)
344  // see https://wiki.panotools.org/Fisheye_Projection
345  Real squaredNorm = sensor.squaredNorm();
346  if (squaredNorm == 0) {
347  // avoid divide-by-zero later
348  return Vector3(0, 0, -1);
349  }
350  Real norm = sqrt(squaredNorm);
351  Real r = undistort(norm);
352  Real theta;
353  if (type == Type::FTHETA) {
354  // r = theta
355  theta = r;
356  } else if (type == Type::RECTILINEAR) {
357  // r = tan(theta)
358  theta = atan(r);
359  } else if (type == Type::EQUISOLID) {
360  // r = 2 sin(theta / 2)
361  theta = 2 * asin(r / 2);
362  } else {
363  CHECK(type == Type::ORTHOGRAPHIC) << "unexpected: " << int(type);
364  // r = sin(theta)
365  theta = asin(r);
366  }
367  Vector3 unit;
368  unit.head<2>() = sin(theta) / norm * sensor;
369  unit.z() = -cos(theta);
370 
371  return unit;
372  }
