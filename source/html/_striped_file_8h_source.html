53  PendingRead* readBegin(uint8_t* dst, uint64_t offset, uint64_t size) const {
54  // check alignment constraints
55  CHECK(align(offset, kStripeSize) == offset);
56  // create vector to store PendingReads
57  const uint64_t stripeCount = align(size, kStripeSize) / kStripeSize;
58  const bool kPerStripe = false; // one read per stripe or one read per disk?
59  const uint64_t readCount = kPerStripe ? stripeCount : disks.size();
60  PendingRead* result = new PendingRead(readCount);
61  // for each disk, compute local disk offset and memory segments
62  std::vector<uint64_t> offsets(disks.size(), UINT64_MAX);
63  std::vector<std::vector<AsyncFile::Segment>> segments(disks.size());
64  for (int stripe = 0; stripe < int(stripeCount); ++stripe) {
65  uint64_t local, disk;
66  calcStripe(local, disk, offset);
67  AsyncFile::Segment segment = {dst, std::min(size, kStripeSize)};
68  if (kPerStripe) {
69  // kick off a separate read for every stripe, don't coalesce per disk
70  disks[disk].readBegin((*result)[stripe], {segment}, local);
71  } else {
72  offsets[disk] = std::min(local, offsets[disk]);
73  segments[disk].push_back(segment);
74  }
75  dst += kStripeSize;
76  offset += kStripeSize;
77  size -= kStripeSize;
78  }
79  if (!kPerStripe) {
80  // kick off one read per disk
81  for (int read = 0; read < int(disks.size()); ++read) {
82  CHECK_NE(offsets[read], UINT64_MAX);
83  disks[read].readBegin((*result)[read], segments[read], offsets[read]);
84  }
