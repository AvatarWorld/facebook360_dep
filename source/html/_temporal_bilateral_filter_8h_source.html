26 // helper for jointBilateralFilter and jointBilateralUpsampling. call one of
27 // those instead of this. when computing the bilateral weight, two colors are
28 // compared. the generalization is that the color for the current pixel comes
29 // from guide, and the colors for the neighbor pixels come from neighborGuide.
30 // For the standard joint-bilateral filter, guide and neighborGuide are the
31 // same, but for the improved joint-bilatral upsampler, guide is the high
32 // resolution image, and neighborGuide is a blurred version of it, which is a
33 // proxy for a low-resolution version of the image.
34 // Pixel should be either float, Vec2f or Vec3f.
35 // guide and neighborGuide should be cv::Mats of type CV_32FC3, CV_16UC3 or CV_8UC3
36 // weightR, weightG, and weightB control how much weight is on each color
37 // channel in computing color differences for bilateral weight.
38 template <typename TPixel, typename TGuide>
39 cv::Mat_<TPixel> generalizedJointBilateralFilter(
40  const cv::Mat_<TPixel>& image, // Either float, Vec2f or Vec3f
41  const cv::Mat_<TGuide>& guide,
42  const cv::Mat_<TGuide>& neighborTGuide,
43  const cv::Mat_<bool>& mask,
44  const int radius,
45  const float sigma,
46  const float weight0 = 1.0f,
47  const float weight1 = 1.0f,
48  const float weight2 = 1.0f,
49  const int numThreads = -1) {
50  CHECK_EQ(guide.size(), neighborTGuide.size());
51  CHECK_EQ(image.size(), guide.size());
52  CHECK_EQ(guide.size(), mask.size());
53 
54  const TPixel zero = 0.0;
55 
56  cv::Mat_<TPixel> dest(image.size());
57  ThreadPool threadPool(numThreads);
58  const int edgeX = image.cols;
59  const int edgeY = 1;
60  for (int yBegin = 0; yBegin < image.rows; yBegin += edgeY) {
61  for (int xBegin = 0; xBegin < image.cols; xBegin += edgeX) {
62  const int xEnd = std::min(xBegin + edgeX, image.cols);
63  const int yEnd = std::min(yBegin + edgeY, image.rows);
64  threadPool.spawn([&, xBegin, yBegin, xEnd, yEnd] {
65  for (int y = yBegin; y < yEnd; ++y) {
66  for (int x = xBegin; x < xEnd; ++x) {
67  if (!mask(y, x)) {
68  dest(y, x) = image(y, x);
69  continue;
70  }
71 
72  const auto guideColor = guide(y, x);
73  float sumWeight = 0.0f;
74  TPixel weightedAvg = zero;
75 
76  const float guideFactor = 1 / cv_util::maxPixelValue(guide);
77  const float neighborTGuideFactor = 1 / cv_util::maxPixelValue(neighborTGuide);
78 
79  for (int v = -radius; v <= radius; ++v) {
80  for (int u = -radius; u <= radius; ++u) {
81  const int sampleX = math_util::clamp(x + u, 0, image.cols - 1);
82  const int sampleY = math_util::clamp(y + v, 0, image.rows - 1);
83 
84  if (!mask(sampleY, sampleX)) {
85  continue;
86  }
87 
88  const TGuide& neighborTGuideColor = neighborTGuide(sampleY, sampleX);
89 
90  const float colorDiffSq = // BGR
91  weight0 *
92  math_util::square(
93  (guideColor[0] * guideFactor) -
94  (neighborTGuideColor[0] * neighborTGuideFactor)) +
95  weight1 *
96  math_util::square(
97  (guideColor[1] * guideFactor) -
98  (neighborTGuideColor[1] * neighborTGuideFactor)) +
99  weight2 *
100  math_util::square(
101  (guideColor[2] * guideFactor) -
102  (neighborTGuideColor[2] * neighborTGuideFactor));
103  const float weight =
104  expf((-colorDiffSq / 3.0f) / (2.0f * math_util::square(sigma)));
105 
106  sumWeight += weight;
107  weightedAvg += weight * image(sampleY, sampleX);
108  }
109  }
110  if (sumWeight != 0.0f) {
111  weightedAvg /= sumWeight;
112  dest(y, x) = weightedAvg;
113  } else {
114  dest(y, x) = image(y, x);
115  }
116  }
117  }
118  });
119  }
120  }
121  threadPool.join();
122  return dest;
123 }

126 static void temporalJointBilateralFilterCol(
127  const std::vector<cv::Mat_<T>>& guides,
128  const std::vector<cv::Mat_<float>>& images,
129  const std::vector<cv::Mat_<bool>>& masks,
130  const int frameOffset,
131  const float sigma,
132  const int spatialRadius,
133  const float weight0,
134  const float weight1,
135  const float weight2,
136  cv::Mat_<float>& result,
137  const int y) {
138  const float maxImageValue = cv_util::maxPixelValue(guides[frameOffset]);
139  for (int x = 0; x < result.cols; ++x) {
140  if (!masks[frameOffset](y, x)) {
141  result(y, x) = images[frameOffset](y, x);
142  continue;
143  }
144  float weightedSumPix = 0.0f;
145  float sumWeight = 0.0f;
146  const T referenceColor = guides[frameOffset](y, x);
147  for (ssize_t t = 0; t < ssize(guides); ++t) {
148  for (int u = -spatialRadius; u <= spatialRadius; ++u) {
149  for (int v = -spatialRadius; v <= spatialRadius; ++v) {
150  const int sampleX = math_util::clamp(x + u, 0, result.cols - 1);
151  const int sampleY = math_util::clamp(y + v, 0, result.rows - 1);
152  if (!masks[t](sampleY, sampleX)) {
153  continue;
154  }
155 
156  const T sampleColor = guides[t](sampleY, sampleX);
157 
158  const float weightedDiff = // BGR
159  weight0 * math_util::square((referenceColor[0] - sampleColor[0]) / maxImageValue) +
160  weight1 * math_util::square((referenceColor[1] - sampleColor[1]) / maxImageValue) +
161  weight2 * math_util::square((referenceColor[2] - sampleColor[2]) / maxImageValue);
162  const float weight = expf(-weightedDiff / math_util::square(sigma));
163 
164  weightedSumPix += images[t](y, x) * weight;
165  sumWeight += weight;
166  }
167  }
168  }
169  result(y, x) = (weightedSumPix / sumWeight);
170  }

173 // temporal joint-bilateral filter: intended for use with time series of depth

193  CHECK_GE(guides[t].channels(), 3);
