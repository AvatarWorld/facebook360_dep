32  const std::vector<Triangle>& triangles,
33  const int thresholdNumTrisForLeaf,
34  const int splitK,
35  const int currDepth,
36  const int maxDepth) {
37  BoundingVolumeHierarchy bvh;
38 
39  // construct a sphere that contains all of the triangles we have. first,
40  // find the center of mass of their vertices. then find a sufficient radius
41  cv::Vec3f cm(0.0f, 0.0f, 0.0f);
42  for (const Triangle& t : triangles) {
43  cm += t.v0 + t.v1 + t.v2;
44  }
45  cm /= float(triangles.size() * 3);
46  bvh.sphere.center = cm;
47 
48  bvh.sphere.radius = 0.0f;
49  for (const Triangle& t : triangles) {
50  bvh.sphere.radius = std::max(bvh.sphere.radius, float(norm(cm - t.v0)));
51  bvh.sphere.radius = std::max(bvh.sphere.radius, float(norm(cm - t.v1)));
52  bvh.sphere.radius = std::max(bvh.sphere.radius, float(norm(cm - t.v2)));
53  }
54 
55  // if we hit one of the termination criteria, stop:
56  // 1. reached max depth
57  // 2. not enough triangles to go further
58  if (currDepth >= maxDepth || int(triangles.size()) < splitK ||
59  int(triangles.size()) < thresholdNumTrisForLeaf) {
60  bvh.isLeaf = true;
61  bvh.leafTriangles = triangles;
62  return bvh;
63  }
64 
65  // at this point we know it's not going to be a leaf. we will pick splitK
66  // triangles at random to be "cluster centers", assign each triangle to
67  // its closest cluster center, and make a child volume around each center.
68  // this is similar to doing 1 iteration of k-means clustering with random
69  // initialization.
70  bvh.isLeaf = false;
71 
72  // pick splitK unique triangles
73  std::vector<int> centerTriangleIndices;
74  std::set<int> centerTriangleSet;
75  while (int(centerTriangleIndices.size()) < splitK) {
76  int r = rand() % triangles.size();
77  if (centerTriangleSet.count(r) == 0) {
78  centerTriangleIndices.push_back(r);
79  centerTriangleSet.insert(r);
80  }
81  }
82 
83  // for each triangle, assign it to its closest cluster center.
84  // closeness will be determined by distance from the first vertex.
85  // this could work badly if there are really unevenly sized triangles.
86  std::vector<int> clusterAssignment(triangles.size());
87  for (int i = 0; i < int(triangles.size()); ++i) {
88  float minDist = FLT_MAX;
89  for (int j = 0; j < splitK; ++j) {
90  const cv::Vec3f clusterJCenter = triangles[centerTriangleIndices[j]].v0;
91  const cv::Vec3f diff = triangles[i].v0 - clusterJCenter;
92  const float dist2 = diff[0] * diff[0] + diff[1] * diff[1] + diff[2] * diff[2];
93  if (dist2 < minDist) {
94  minDist = dist2;
95  clusterAssignment[i] = j;
96  }
97  }
98  }
99 
100  // for each cluster, recursively make a child bvh containing the triangles
101  // that were assigned to that cluster.
102  std::vector<std::vector<Triangle>> trianglesInCluster(splitK);
103  for (int i = 0; i < int(triangles.size()); ++i) {
104  trianglesInCluster[clusterAssignment[i]].push_back(triangles[i]);
105  }
106 
107  for (int j = 0; j < splitK; ++j) {
108  bvh.children.push_back(
109  makeBVH(trianglesInCluster[j], thresholdNumTrisForLeaf, splitK, currDepth + 1, maxDepth));
110  }
111

113  }
