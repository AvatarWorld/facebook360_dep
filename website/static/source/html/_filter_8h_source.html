42  cv::Mat_<cv::Vec3f> buffer(std::max(inputImage.rows, inputImage.cols), 1);
43 
44  // Horizontal pass
45  for (int i = 0; i < lpImage.rows; ++i) {
46  // Causal pass
47  cv::Vec3f v(inputImage(i, lpImage.cols - 1));
48  for (int j = 0; j < lpImage.cols; ++j) {
49  cv::Vec3f ip(inputImage(i, j));
50  v = math_util::lerp(ip, v, alpha);
51  buffer(hBoundary(j - 1, lpImage.cols), 0) = v;
52  }
53 
54  // Anticausal pass
55  v = buffer(0, 0);
56  for (int j = lpImage.cols - 1; j >= 0; --j) {
57  cv::Vec3f ip(buffer(math_util::wrap(j, lpImage.cols), 0));
58  v = math_util::lerp(ip, v, alpha);
59  lpImage(i, hBoundary(j + 1, lpImage.cols))[0] = math_util::clamp(v[0], 0.0f, maxVal);
60  lpImage(i, hBoundary(j + 1, lpImage.cols))[1] = math_util::clamp(v[1], 0.0f, maxVal);
61  lpImage(i, hBoundary(j + 1, lpImage.cols))[2] = math_util::clamp(v[2], 0.0f, maxVal);
62  }
63  }
64 
65  // Vertical pass
66  for (int j = 0; j < lpImage.cols; ++j) {
67  // Causal pass
68  cv::Vec3f v(lpImage(1, j));
69  for (int i = 0; i < lpImage.rows; ++i) {
70  cv::Vec3f ip(lpImage(i, j));
71  v = math_util::lerp(ip, v, alpha);
72  buffer(vBoundary(i - 1, lpImage.rows), 0) = v;
73  }
74  // Anticausal pass
75  v = buffer(lpImage.rows - 2, 0);
76  for (int i = lpImage.rows - 1; i >= -1; --i) {
77  cv::Vec3f ip = buffer(math_util::reflect(i, lpImage.rows), 0);
78  v = math_util::lerp(ip, v, alpha);
79  lpImage(vBoundary(i + 1, lpImage.rows), j)[0] = math_util::clamp(v[0], 0.0f, maxVal);
