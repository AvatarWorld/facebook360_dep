128  void demosaicFrequencyFilter(cv::Mat_<float>& r, cv::Mat_<float>& g, cv::Mat_<float>& b) const {
129  // Green/luma 4-th order Butterworth lp filter
130  const math_util::Butterworth dFilter(0.0f, 2.0f, width + height, 1.0f, 4.0);
131  // Chrome cross over filter
132  const math_util::Butterworth dcFilter(0.0f, 2.0f, width + height, 1.0f, 2.0f);
133 
134  // Do a per pixel filtering in DCT space
135  const int h2 = r.rows;
136  const int w2 = r.cols;
137  for (int i = 0; i < h2; ++i) {
138  const float y = float(i) / float(h2 - 1);
139  for (int j = 0; j < w2; ++j) {
140  const float x = float(j) / float(w2 - 1);
141  // Diagonal distance and half
142  static const float kDScale = 1.2f;
143  const float d = (x + y) * kDScale;
144  const float kSharpen = d / 2.5f + 1.0f;
145  const float gGain = 2.0f * dFilter(d) * kSharpen;
146  const float rbGain = 4.0f * dFilter(d);
147  g(i, j) *= gGain;
148 
149  const float kCrossoverCutoff = 3.0f;
150  const float d2 = d * 2 * kCrossoverCutoff;
151 
152  // Cross over blend value
153  const float alpha = dcFilter(d2);
154  r(i, j) = math_util::lerp(g(i, j), r(i, j) * rbGain, alpha);
155  b(i, j) = math_util::lerp(g(i, j), b(i, j) * rbGain, alpha);
156  }

380  // Build the composite tone curve map from [0,1]^3 to [0,1]^3
381  void buildToneCurveLut() {
382  toneCurveLut.clear();
383  const float dx = 1.0f / float(kToneCurveLutSize - 1);
384 
385  // Contrast angle constants
386  const float angle = M_PI * 0.25f * contrast;
387  const float slope = tanf(angle);
388  const float bias = 0.5f * (1.0f - slope);
389 
390  for (int i = 0; i < kToneCurveLutSize; ++i) {
391  const float x = dx * i;
392  if (!toneCurveEnabled) {
393  // Just a linear ramp ==> no-op
394  const float y = x;
395  toneCurveLut.push_back(cv::Vec3f(y, y, y));
396  } else {
397  // Apply gamma correction
398  float r = powf(x, gamma.x);
399  float g = powf(x, gamma.y);
400  float b = powf(x, gamma.z);
401  // Then low/high key boost
402  r = lowKey(lowKeyBoost.x, r) + highKey(highKeyBoost.x, r);
403  g = lowKey(lowKeyBoost.y, g) + highKey(highKeyBoost.y, g);
404  b = lowKey(lowKeyBoost.z, b) + highKey(highKeyBoost.z, b);
405 
406  // Then contrast
407  r = math_util::clamp((slope * r + bias), 0.0f, 1.0f);
408  g = math_util::clamp((slope * g + bias), 0.0f, 1.0f);
409  b = math_util::clamp((slope * b + bias), 0.0f, 1.0f);
410 
411  // Place it in the table.
412  toneCurveLut.push_back(cv::Vec3f(r, g, b));
413  }
414  }

763  /* Load sensorWidth x sensorHeight image from raw data, interleaving, byte swapping, and
764  re-indexing if necessary for host and OpenCV compatibility. */
765  template <typename T>
766  void loadImageFromSensor(const std::vector<T>& inputImage) {
767  // - sanity check config
768  CHECK_NE(sensorWidth, 0);
769  CHECK_NE(sensorHeight, 0);
770  CHECK_EQ(sensorWidth % 2, 0);
771  CHECK_EQ(sensorHeight % 2, 0);
772  CHECK_EQ(inputImage.size(), sensorWidth * sensorHeight);
773  CHECK_EQ(sensorBitsPerPixel, 8 * sizeof(T));
774 
775  cv::Mat_<T> intermediateImage(sensorHeight, sensorWidth);
776 
777  // - load directly if there's nothing to do
778  if (!isPlanar && isLittleEndian == folly::kIsLittleEndian && isRowMajor) {
779  memcpy(intermediateImage.data, inputImage.data(), sensorWidth * sensorHeight * sizeof(T));
780  loadImage(intermediateImage);
781  return;
782  }
783 
784  // - setup problem dimensions and channel mappings
785  int planeWidth = sensorWidth / 2;
786  int planeHeight = sensorHeight / 2;
787  int numPixelsPerPlane = planeWidth * planeHeight;
788  std::array<int, 4> inputChannelOrderToBayerOrder = {
789  {0, 1, 2, 3}}; // no-op for interleaved input
790  if (isPlanar) {
791  inputChannelOrderToBayerOrder = getPlaneOrderToBayerOrder();
792  }
793 
794  // - inner loop helpers to byte swap, interleave, and re-index as needed
795  // - converts from linear index to a representation-independent (channel, row, column)
796  // pixel index
797  auto pixelIndexFromInputLinearIndex = [=](int linearIndex) {
798  int channel, row, col;
799  if (isPlanar) {
800  channel = linearIndex / numPixelsPerPlane;
801  int linearIndexInPlane = linearIndex - channel * numPixelsPerPlane;
802  std::tie(row, col) =
803  math_util::linearToMatrixIndex(linearIndexInPlane, planeHeight, planeWidth, isRowMajor);
804  } else {
805  std::tie(row, col) =
806  math_util::linearToMatrixIndex(linearIndex, sensorHeight, sensorWidth, isRowMajor);
807  // Bayer pattern is always specified in row major, so compute the channel index accordingly
808  channel = math_util::matrixToLinearIndex(std::make_tuple(row % 2, col % 2), 2, 2);
809  row /= 2;
810  col /= 2;
811  }
812  return std::make_tuple(channel, row, col);
813  };
814 
815  // - converts representation-independent (channel, row, column) pixel index to (row, column)
816  // matrix index in interleaved image
817  auto outputInterleavedRawIndexFromPixelIndex = [=](std::tuple<int, int, int> pixelIndex) {
818  int channel, row, col, bayerRow, bayerCol;
819  std::tie(channel, row, col) = pixelIndex;
820  channel = inputChannelOrderToBayerOrder[channel];
821  std::tie(bayerRow, bayerCol) = math_util::linearToMatrixIndex(channel, 2, 2);
822  return std::make_pair(2 * row + bayerRow, 2 * col + bayerCol);
823  };
824 
825  auto byteSwap = isLittleEndian ? folly::Endian::little<T> : folly::Endian::big<T>;
826 
827  // - reorder and load image
828  for (int k = 0; k < sensorWidth * sensorHeight; ++k) {
829  auto pixelIndex = pixelIndexFromInputLinearIndex(k);
830  auto rawIndex = outputInterleavedRawIndexFromPixelIndex(pixelIndex);
831  intermediateImage(std::get<0>(rawIndex), std::get<1>(rawIndex)) = byteSwap(inputImage[k]);
832  }
833  loadImage(intermediateImage);
834  }

849  CHECK(false) << "input is larger that 16 bits per pixel";
850  }

979  void removeStuckPixels() {
980  if (stuckPixelRadius > 0) {
981  struct Pval {
982  float val;
983  int i;
984  int j;
985 
986  Pval(float v) : val(v) {}
987 
988  bool operator<(const Pval& p) const {
989  return val < p.val;
990  }
991 
992  Pval& operator=(const Pval& p) {
993  val = p.val;
994  i = p.i;
995  j = p.j;
996  return *this;
997  }
998  };
999 
1000  std::vector<Pval> region;
1001 
1002  for (int i = 0; i < height; ++i) {
1003  // Traverse boustrophedonically
1004  const bool evenScanLine = (i % 2) == 0;
1005  const int jStart = evenScanLine ? 0 : width - 1;
1006  const int jEnd = evenScanLine ? width - 1 : 0;
1007  const int jStep = evenScanLine ? 1 : -1;
1008 
1009  for (int j = jStart; j != jEnd; j += jStep) {
1010  const bool thisPixelRed = redPixel(i, j);
1011  const bool thisPixelGreen = greenPixel(i, j);
1012  const bool thisPixelBlue = bluePixel(i, j);
1013 
1014  region.clear();
1015 
1016  float mean = 0.0f;
1017  for (int y = -stuckPixelRadius; y <= stuckPixelRadius; y++) {
1018  const int ip = math_util::reflect(i + y, height);
1019  for (int x = -stuckPixelRadius; x <= stuckPixelRadius; x++) {
1020  const int jp = math_util::reflect(j + x, width);
1021  Pval p(rawImage(ip, jp));
1022  p.i = ip;
1023  p.j = jp;
1024  if (redPixel(ip, jp) && thisPixelRed) {
1025  mean += p.val;
1026  region.push_back(p);
1027  } else if (greenPixel(ip, jp) && thisPixelGreen) {
1028  mean += p.val;
1029  region.push_back(p);
1030  } else if (bluePixel(ip, jp) && thisPixelBlue) {
1031  mean += p.val;
1032  region.push_back(p);
1033  }
1034  }
1035  }
1036  mean /= float(region.size());
1037 
1038  // Only deal with dark regions
1039  if (mean < stuckPixelDarknessThreshold) {
1040  sort(region.begin(), region.end());
1041 
1042  // See if the middle pixel is an outlier
1043  for (int k = int(region.size()) - 1; k >= int(region.size()) - stuckPixelThreshold;
1044  --k) {
1045  if (region[k].i == i && region[k].j == j) {
1046  // Replace the pixel pixel with the median of the region.
1047  rawImage(i, j) = region[region.size() / 2].val;
1048  goto l0;
1049  }
1050  }
1051  }
1052  l0:
1053  continue;
1054  }
1055  }
1056  }
1057  }

1059  void blackLevelAdjust() {
1060  const float br = blackLevel.x;
1061  const float bg = blackLevel.y;
1062  const float bb = blackLevel.z;
1063  const float sr = 1.0f / (1.0f - br);
1064  const float sg = 1.0f / (1.0f - bg);
1065  const float sb = 1.0f / (1.0f - bb);
1066  for (int i = 0; i < height; ++i) {
1067  for (int j = 0; j < width; j++) {
1068  if (rawImage(i, j) < 1.0f) {
1069  if (redPixel(i, j)) {
1070  rawImage(i, j) = (rawImage(i, j) - br) * sr;
1071  } else if (greenPixel(i, j)) {
1072  rawImage(i, j) = (rawImage(i, j) - bg) * sg;
1073  } else {
1074  rawImage(i, j) = (rawImage(i, j) - bb) * sb;
1075  }
1076  }
1077  }
1078  }
