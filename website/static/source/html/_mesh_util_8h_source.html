24 Eigen::Vector3d calcBarycentrics(const Eigen::Vector2d& point, const Eigen::Matrix3d& triangle) {
25  const Eigen::Vector2d base = triangle.row(2).head<2>();
26  Eigen::Matrix2d m = triangle.topLeftCorner<2, 2>();
27  m.row(0) -= base;
28  m.row(1) -= base;
29  Eigen::Vector3d result;
30  result.head<2>() = m.transpose().colPivHouseholderQr().solve(point - base);
31  result.z() = 1 - result.x() - result.y();
32  return result;
33 }

35 inline void writePfm(
36  const cv::Mat_<float>& original,
37  const Camera::Vector2& resolution,
38  const Eigen::MatrixXd& vertexes,
39  const Eigen::MatrixXi& faces,
40  const filesystem::path& filenamePfm) {
41  LOG(INFO) << "Writing PFM file...";
42 
43  // rasterize each face into dst
44  cv::Mat_<float> dst(original.size(), -FLT_MAX);
45  for (int face = 0; face < faces.rows(); ++face) {
46  Eigen::Matrix3d triangle;
47  for (int i = 0; i < triangle.rows(); ++i) {
48  triangle.row(i) = vertexes.row(faces(face, i));
49  }
50  // rescale x,y to original depth map's size
51  triangle.col(0) *= original.cols / resolution.x();
52  triangle.col(1) *= original.rows / resolution.y();
53  // crude rasterizer
54  Eigen::Vector3d bboxMin = triangle.colwise().minCoeff();
55  Eigen::Vector3d bboxMax = triangle.colwise().maxCoeff();
56  // for each pixel center in bounding box
57  for (int y = floor(bboxMin.y()); y < ceil(bboxMax.y()); ++y) {
58  for (int x = floor(bboxMin.x()); x < ceil(bboxMax.x()); ++x) {
59  // ignore rasterization rules, just include all edges
60  Eigen::Vector3d bary = calcBarycentrics({x + 0.5, y + 0.5}, triangle);
61  if (bary.x() >= 0 && bary.y() >= 0 && bary.z() >= 0) {
62  CHECK(0 <= x && x < dst.cols) << x << dst.cols;
63  CHECK(0 <= y && y < dst.rows) << y << dst.rows;
64  dst(y, x) = float(triangle.col(2).dot(bary));
65  }
66  }
67  }
68  }
69  cv_util::writeCvMat32FC1ToPFM(filenamePfm, dst);
70 }

166 // return a mask representing which of the 4 possible triangles to output
167 inline unsigned getTriangleMask(
168  const Eigen::MatrixXd& verts,
169  const int base,
170  const int width,
171  const float tearRatio,
172  const bool isRigCoordinates) {
173  const int tli = base;
174  const int tri = base + 1;
175  const int bli = base + width;
176  const int bri = base + width + 1;
177 
178  // If isRigCoordinates, distance from rig center is the norm of the vector
179  // If individual camera, z-coord is distance from rig center
180  const double tl = isRigCoordinates ? verts.row(tli).norm() : verts(tli, 2);
181  const double tr = isRigCoordinates ? verts.row(tri).norm() : verts(tri, 2);
182  const double bl = isRigCoordinates ? verts.row(bli).norm() : verts(bli, 2);
183  const double br = isRigCoordinates ? verts.row(bri).norm() : verts(bri, 2);
184 
185  std::vector<std::tuple<double, int>> v = {std::make_tuple(tl, 0),
186  std::make_tuple(tr, 1),
187  std::make_tuple(bl, 2),
188  std::make_tuple(br, 3)};
189 
190  sort(v.begin(), v.end());
191 
192  // are all 4 values pretty close?
193  if (std::get<0>(v.front()) / std::get<0>(v.back()) > tearRatio) {
194  // output both triangles, splitting along the shallowest diagonal
195  if (std::abs(tl - br) < std::abs(tr - bl)) {
196  return 1 << 1 | 1 << 2; // triangles 1 and 2
197  }
198  return 1 << 0 | 1 << 3; // triangles 0 and 3
199  }
200  double lo = std::get<0>(v.front()) / std::get<0>(v[2]);
201  double hi = std::get<0>(v[1]) / std::get<0>(v.back());
202 
203  // are the 3 lowest values pretty close?
204  if (lo >= tearRatio && lo > hi) {
205  // output the triangle that does not include back
206  int index = std::get<1>(v.back()) ^ 0x3;
207  return 1 << index;
208  }
209 
210  // are the 3 highest values pretty close?
211  if (hi >= tearRatio) {
212  // output the triangle that does not include front
213  int index = std::get<1>(v.front()) ^ 0x3;
214  return 1 << index;
215  }
216 
217  // don't output anything
218  return 0;
219 }

248 // Gets a set of vertexes and creates corresponding faces
249 // wrapHorizontally and isRigCoordinates explain the semantics of vertexes
250 // - wrapHorizontally: true to fix (link) meridian ends on equirect
251 // - isRigCoordinates: true if vertexes hold plain rig coords,
252 // false if vertex z represents distance from rig center
253 // examples of (wrapHorizontally, isRigCoordinates):
254 // - rig coordinate equirect mesh = (true, true)
255 // - rig coordinate camera mesh = (false, true)
256 // - equierror camera mesh = (false, false)
257 //
258 // tearRatio causes slivery triangles to be discarded if
259 // min(depth) / max(depth) < tearRatio
260 //
261 // a reasonable value to try is ~0.95, which means it won't connect
262 // vertexes if one is at 10 m while the neighbor is at 9.5 m
263 inline Eigen::MatrixXi getFaces(
264  const Eigen::MatrixXd& vertexes,
265  const int width,
266  const int height,
267  const bool wrapHorizontally,
268  const bool isRigCoordinates,
269  const float tearRatio = 0.0f) {
270  Eigen::MatrixXi faces(width * (height - 1) * 2, 3);
271  int face = 0;
272  for (int y = 0; y < height - 1; ++y) {
273  for (int x = 0; x < width - 1; ++x) {
274  int base = y * width + x;
275  unsigned mask = getTriangleMask(vertexes, base, width, tearRatio, isRigCoordinates);
276  for (int triangle = 0; triangle < 4; ++triangle) {
277  if ((mask >> triangle) & 1) {
278  addTriangle(faces.row(face++), triangle, base, width);
279  }
280  }
281  }
282  }
283 
284  if (wrapHorizontally) {
285  // Link last and first longitudes
286  // Note how triangles are always defined counterclock-wise
287  for (int y = 0; y < height - 1; ++y) {
288  int base = y * width;
289  faces.row(face++) = Eigen::Vector3i(base + width, base, base + width - 1);
290  faces.row(face++) = Eigen::Vector3i(base + width - 1, base + 2 * width - 1, base + width);
291  }
292  }
293 
294  return faces.topRows(face);
295 }

297 inline Eigen::MatrixXd getVertexesEquirect(const cv::Mat_<float>& disparity, const float maxDepth) {
298  const int width = disparity.cols;
299  const int height = disparity.rows;
300  Eigen::MatrixXd vertexes(width * height, 3);
301  for (int y = 0; y < height; ++y) {
302  for (int x = 0; x < width; ++x) {
303  const float u = float(x + 0.5) / float(width);
304  const float v = float(y + 0.5) / float(height);
305  const float theta = u * 2.0f * M_PI;
306  const float phi = v * M_PI;
307  const float depth = std::fmin(maxDepth, 1.0f / disparity(y, x));
308  vertexes.row(y * width + x) =
309  depth * Eigen::Vector3d(sin(phi) * cos(theta), cos(phi), sin(phi) * sin(theta));
310  }
311  }
312  return vertexes;
313 }

342 // Assumes vertexes have been generated in order from a depth map stored row-major so that
343 // mask(y, x) corresponds to vertexes(y * mask.cols + x)
344 inline void applyMaskToVertexesAndFaces(
345  Eigen::MatrixXd& vertexes,
346  Eigen::MatrixXi& faces,
347  const cv::Mat_<bool> mask) {
348  const int width = mask.cols;
349  const int height = mask.rows;
350  CHECK_EQ(width * height, vertexes.rows());
351  bool* originalVertexMask = (bool*)mask.data;
352 
353  // Keep faces only if all vertexes have a non-zero mask
354  std::vector<int> outputFaceIndexes;
355  for (int i = 0; i < faces.rows(); ++i) {
356  const Eigen::Vector3i& face = faces.row(i);
357  bool allVertexesRetained = true;
358  for (int j = 0; j < face.size(); ++j) {
359  if (!originalVertexMask[face(j)]) {
360  allVertexesRetained = false;
361  break;
362  }
363  }
364  if (allVertexesRetained) {
365  outputFaceIndexes.push_back(i);
366  }
367  }
368 
369  // Keep only vertexes of retained faces
370  std::vector<bool> vertexMask(vertexes.rows(), false);
371  for (int i : outputFaceIndexes) {
372  const Eigen::Vector3i& face = faces.row(i);
373  for (int j = 0; j < face.size(); ++j) {
374  vertexMask[face(j)] = true;
375  }
376  }
377 
378  // Compute mapping between old and new vertex indices
379  std::map<int, int> inputToOutputVertexIndexes;
380  int numOutputVertexes = 0;
381  for (int i = 0; i < vertexes.rows(); ++i) {
382  if (vertexMask[i]) {
383  inputToOutputVertexIndexes[i] = numOutputVertexes++;
384  }
385  }
386 
387  // create output vertex matrix
388  Eigen::MatrixXd inputVertexes = vertexes;
389  vertexes = Eigen::MatrixXd(numOutputVertexes, 3);
390  for (auto iter = inputToOutputVertexIndexes.begin(); iter != inputToOutputVertexIndexes.end();
391  ++iter) {
392  vertexes.row(iter->second) = inputVertexes.row(iter->first);
393  }
394 
395  // Create output face matrix and re-index vertexes
396  Eigen::MatrixXi inputFaces = faces;
397  faces = Eigen::MatrixXi(outputFaceIndexes.size(), 3);
398  for (ssize_t i = 0; i < ssize(outputFaceIndexes); ++i) {
399  const Eigen::Vector3i& face = inputFaces.row(outputFaceIndexes[i]);
400  for (int j = 0; j < face.size(); ++j) {
401  faces(i, j) = inputToOutputVertexIndexes[face(j)];
402  }
403  }
404 }

407 inline void addTextureCoordinatesEquirect(Eigen::MatrixXd& vertexes) {
408  vertexes.conservativeResize(Eigen::NoChange, 5);
409  for (int v = 0; v < vertexes.rows(); ++v) {
410  // similar to createEquirectProgram() in ParallaxViewer.cpp:
411  // texture goes +x, +z, -x, -z, +x from left to right (0 to 1)
412  // and -y to +y from top to bottom (0 to 1)
413  Eigen::Vector3d pos = vertexes.row(v).head<3>();
414  const double xzNorm = Eigen::Vector2d(pos.x(), pos.z()).norm();
415  vertexes(v, 3) = std::atan2(-pos.z(), -pos.x()) * 0.5 / M_PI + 0.5;

417  }
