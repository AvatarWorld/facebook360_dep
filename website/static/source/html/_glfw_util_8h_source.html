140  virtual void updateTransform() {
141  const Eigen::Vector3f forward = {
142  std::sin(pitch) * std::cos(yaw), std::sin(pitch) * std::sin(yaw), std::cos(pitch)};
143  const Eigen::Vector3f right = up.cross(-forward).normalized();
144 
145  transform.linear().row(0) = right;
146  transform.linear().row(1) = right.cross(forward);
147  transform.linear().row(2) = -forward;
148  transform.translation() = origin * scale;
149  }

174  virtual void motion(const double x, const double y) {
175  if (mouseAction == GLFW_PRESS) {
176  if (mouseButton == GLFW_MOUSE_BUTTON_LEFT) { // rotate
177  const float dPitch = (mousePos.y() - static_cast<float>(y));
178  const float dYaw = (mousePos.x() - static_cast<float>(x));
179  pitch += (dPitch / height) * (static_cast<float>(M_PI) / 2.0f);
180  yaw -= (dYaw / width) * (static_cast<float>(M_PI) / 2.0f);
181  } else { // pan
182  Eigen::Vector3f move(
183  static_cast<float>(x) - mousePos.x(),
184  mousePos.y() - static_cast<float>(y),
185  0); // y is down
186  origin += move;
187  }
188  updateTransform();
189  }
190  mousePos = {x, y};
191  }

409  static void mainLoop() {
410  // This is not re-entrant so we'll only let one in.
411  static std::mutex once;
412  static std::lock_guard<std::mutex> onlyOnce(once);
413 
414  // Loop until all the windows exit
415  static bool windowsAvailable;
416 
417  // Make sure we read the windows data structure in a thread safe manner.
418  {
419  std::lock_guard<std::mutex> windowLock(windowMapMutex);
420  windowsAvailable = !windows.empty();
421  }
422 
423  // Loop until we kill all the windows
424  while (windowsAvailable) {
425  // This lock allows thread safe window creation and destruction
426  // to happen between window updates.
427  std::lock_guard<std::mutex> windowLock(windowMapMutex);
428 
429  // Loop over all the windows giving each a chance to draw.
430  for (std::map<GLFWwindow*, GlWindow*>::iterator it = windows.begin(); it != windows.end();) {
431  // Check for input between window draws
432  glfwPollEvents();
433 
434  // Grab the window ptrs
435  GLFWwindow* glfwWindow = it->first;
436  GlWindow* glWindow = it->second;
