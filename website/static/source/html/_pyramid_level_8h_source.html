417  void saveDstImage(const int dstIdx, const ImageType imageType, const float scale = 1.0f) {
418  cv::Mat dstImage;
419  switch (imageType) {
420  case ImageType::disparity_levels:
421  dstImage = dstDisparity(dstIdx);
422  break;
423  case ImageType::cost:
424  dstImage = dstCost(dstIdx);
425  break;
426  case ImageType::confidence:
427  dstImage = dstConfidence(dstIdx);
428  break;
429  case ImageType::mismatches:
430  dstImage = overlayMismatchedDstDisparityMask(dstIdx);
431  break;
432  default:
433  CHECK(false) << "unexpected image type " << imageTypes[(int)imageType];
434  }
435 
436  if (dstImage.empty()) {
437  return;
438  }
439 
440  cv::Mat scaledDstImage = dstImage * scale;
441  if (imageType == ImageType::disparity_levels) {
442  // note: disparity values will be clamped to the [0,1] range (which get scaled
443  // to [0, 2^16 - 1]) and nans will be converted to zero
444  scaledDstImage = cv_util::convertTo<uint16_t>(scaledDstImage);
445  }
446 
447  const std::string& dstId = rigDst[dstIdx].id;
448  const filesystem::path fn =
449  depth_estimation::genFilename(outputDir, imageType, level, dstId, frameName, "png");
450  cv_util::imwriteExceptionOnFail(fn, scaledDstImage);
451  }

462  cv::Mat_<cv::Vec4f> overlayMismatchedDstDisparityMask(const int dstIdx) {
463  const cv::Mat_<bool>& mask = dstMismatchedDisparityMask(dstIdx);
464  const cv::Mat_<float>& disparity = dstDisparity(dstIdx);
465  const cv::Mat_<bool>& fov = dstFovMask(dstIdx);
466 
467  cv::Mat_<cv::Vec4f> maskedDisparity(disparity.size(), NAN);
468 
469  for (int x = 0; x < maskedDisparity.cols; ++x) {
470  for (int y = 0; y < maskedDisparity.rows; ++y) {
471  if (!fov(y, x)) {
472  continue;
473  }
474  if (mask(y, x)) {
475  maskedDisparity(y, x) = cv::Vec4f(0.f, 0.f, 1.f, 1.f); // red
476  } else {
477  float d = disparity(y, x);
478  maskedDisparity(y, x) = cv::Vec4f(d, d, d, 1.f);
479  }
480  }
481  }
482 
483  return maskedDisparity;

486  void saveResults(const std::string& outputFormatsStr) {
487  std::vector<std::string> outputFormatsVec;
488  folly::split(",", outputFormatsStr, outputFormatsVec);
489  std::unordered_set<std::string> outputFormats(outputFormatsVec.begin(), outputFormatsVec.end());
490 
491  const bool saveExr = (outputFormats.find("exr") != outputFormats.end());
492  const bool savePfm = true; // always save PFM
493  const bool savePng = (outputFormats.find("png") != outputFormats.end());
494 
495  if (!(saveExr || savePfm || savePng)) {
496  return;
497  }
498  ThreadPool threadPool(numThreads);
499  for (int dstIdx = 0; dstIdx < int(rigDst.size()); ++dstIdx) {
500  threadPool.spawn([&, dstIdx] {
501  const cv::Mat_<float>& disp = dstDisparity(dstIdx);
502  const ImageType imageType = ImageType::disparity_levels;
503 
504  std::map<std::string, bool> types = {{"exr", saveExr}, {"pfm", savePfm}, {"png", savePng}};
505  for (std::pair<std::string, bool> type : types) {
506  if (!type.second) {
507  continue;
508  }
509  const std::string& t = type.first;
510  const std::string& dstId = rigDst[dstIdx].id;
511  const filesystem::path fn =
512  depth_estimation::genFilename(outputDir, imageType, level, dstId, frameName, t);
513  boost::filesystem::create_directories(fn.parent_path());
514  if (t == "exr") {
515  cv_util::imwriteExceptionOnFail(fn, disp);
516  } else if (t == "pfm") {
517  cv_util::writeCvMat32FC1ToPFM(fn, disp);
518  } else if (t == "png") {
519  const cv::Mat_<uint16_t> disp16 = cv_util::convertTo<uint16_t>(disp);
520  cv_util::imwriteExceptionOnFail(fn, disp16);
521  } else {
522  CHECK(false) << "Invalid type: " << t;
523  }
524  }
525  });
526  }
527  threadPool.join();
528  }
